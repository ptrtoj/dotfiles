*** Startup Hook
#+begin_src elisp
  (defun jeon/display-startup-time ()
    (message "Emacs loaded in %s with %d garbage collections."
	     (format "%.2f seconds"
		     (float-time
		      (time-subtract after-init-time before-init-time)))
	     gcs-done))

  (add-hook 'emacs-startup-hook #'jeon/display-startup-time)
#+end_src
